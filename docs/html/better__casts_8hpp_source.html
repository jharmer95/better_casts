197 template<typename T>
198 NODISCARD constexpr auto is_nan(T val) noexcept -> bool
199 {
200 static_assert(std::is_floating_point<T>::value, "T must be floating point to check for NaN");
201
202 // ReSharper disable once CppIdenticalOperandsInBinaryExpression
203 // NOLINTNEXTLINE(misc-redundant-expression)
204 return val != val;
205 }
206
207 static_assert(is_nan(NAN), "NAN must be NAN");
208 static_assert(is_nan(std::numeric_limits<float>::quiet_NaN()), "qNAN must be NAN");
209 static_assert(is_nan(std::numeric_limits<float>::signaling_NaN()), "sNAN must be NAN");
210 static_assert(!is_nan(0.0f), "A valid float must not be NAN");
211 static_assert(!is_nan(INFINITY), "INFINITY must not be NAN");

213 template<typename T>
214 NODISCARD constexpr auto is_inf(T val) noexcept -> bool
215 {
216 static_assert(std::is_floating_point<T>::value, "T must be floating point to check for Infinity");
217
218 // NOLINTNEXTLINE(clang-diagnostic-float-equal)
219#ifdef __clang__
220# pragma clang diagnostic push
221# pragma clang diagnostic ignored "-Wfloat-equal"
222#endif
223 return val == std::numeric_limits<T>::infinity() || val == -std::numeric_limits<T>::infinity();
224#ifdef __clang__
225# pragma clang diagnostic pop
226#endif
227 }
228
229 static_assert(is_inf(INFINITY), "is_inf(INFINITY) must be true");
230 static_assert(is_inf(-INFINITY), "is_inf(-INFINITY) must be true");

258 template<typename To, typename From,
259 typename = std::enable_if_t<std::is_arithmetic<To>::value && std::is_floating_point<From>::value>>
260 constexpr auto round(From val) noexcept -> To
261 {
262 if (val >= float_const<From>::ZERO)
263 {
264 return abs(trunc(val) - val) >= abs(trunc(val) - val + float_const<From>::ONE)
265 ? static_cast<To>(val) + static_cast<To>(1)
266 : static_cast<To>(val);
267 }
268
269 return abs(trunc(val) - val) >= abs(trunc(val) - val - float_const<From>::ONE)
270 ? static_cast<To>(val) - static_cast<To>(1)
271 : static_cast<To>(val);
272 }

274 template<typename To, typename From,
275 typename = std::enable_if_t<std::is_arithmetic<To>::value && std::is_floating_point<From>::value>>
276 constexpr auto floor(From val) noexcept -> To
277 {
278#ifdef __clang__
279# pragma clang diagnostic push
280# pragma clang diagnostic ignored "-Wfloat-equal"
281#endif
282 if (trunc(val) == val)
283 {
284 return static_cast<To>(val);
285 }
286#ifdef __clang__
287# pragma clang diagnostic pop
288#endif
289
290 return val < float_const<From>::ZERO ? static_cast<To>(val) - static_cast<To>(1) : static_cast<To>(val);
291 }
