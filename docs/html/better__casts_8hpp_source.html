190 template<typename T>
191 NODISCARD constexpr auto is_nan(T val) noexcept -> bool
192 {
193 static_assert(std::is_floating_point<T>::value, "T must be floating point to check for NaN");
194
195 // ReSharper disable once CppIdenticalOperandsInBinaryExpression
196 // NOLINTNEXTLINE(misc-redundant-expression)
197 return val != val;
198 }
199
200 static_assert(is_nan(NAN), "NAN must be NAN");
201 static_assert(is_nan(std::numeric_limits<float>::quiet_NaN()), "qNAN must be NAN");
202 static_assert(is_nan(std::numeric_limits<float>::signaling_NaN()), "sNAN must be NAN");
203 static_assert(!is_nan(0.0f), "A valid float must not be NAN");
204 static_assert(!is_nan(INFINITY), "INFINITY must not be NAN");
205
206 template<typename T>
207 NODISCARD constexpr auto is_inf(T val) noexcept -> bool
208 {
209 static_assert(std::is_floating_point<T>::value, "T must be floating point to check for Infinity");
210
211 // NOLINTNEXTLINE(clang-diagnostic-float-equal)
212#ifdef __clang__
213# pragma clang diagnostic push
214# pragma clang diagnostic ignored "-Wfloat-equal"
215#endif
216 return val == std::numeric_limits<T>::infinity() || val == -std::numeric_limits<T>::infinity();
217#ifdef __clang__
218# pragma clang diagnostic pop
219#endif
220 }
221
222 static_assert(is_inf(INFINITY), "is_inf(INFINITY) must be true");
223 static_assert(is_inf(-INFINITY), "is_inf(-INFINITY) must be true");
224
225 template<typename T>
226 constexpr void check_inf_nan(T val)
227 {
228 if (is_nan(val))
229 {
230 throw float_cast_error("float_cast failed: cannot cast from NaN");

232
233 if (is_inf(val))
234 {
235 throw float_cast_error("float_cast failed: cannot cast from Infinity");

251 template<typename To, typename From,
252 typename = std::enable_if_t<std::is_arithmetic<To>::value && std::is_floating_point<From>::value>>
253 constexpr auto round(From val) noexcept -> To
254 {
255 if (val >= float_const<From>::ZERO)
256 {
257 return abs(trunc(val) - val) >= abs(trunc(val) - val + float_const<From>::ONE)
258 ? static_cast<To>(val) + static_cast<To>(1)
259 : static_cast<To>(val);
260 }
261
262 return abs(trunc(val) - val) >= abs(trunc(val) - val - float_const<From>::ONE)
263 ? static_cast<To>(val) - static_cast<To>(1)
264 : static_cast<To>(val);
265 }
266
267 template<typename To, typename From,
268 typename = std::enable_if_t<std::is_arithmetic<To>::value && std::is_floating_point<From>::value>>
269 constexpr auto floor(From val) noexcept -> To
270 {
271#ifdef __clang__
272# pragma clang diagnostic push
273# pragma clang diagnostic ignored "-Wfloat-equal"
274#endif
275 if (trunc(val) == val)
276 {
277 return static_cast<To>(val);
278 }
279#ifdef __clang__
280# pragma clang diagnostic pop
281#endif
282
283 return val < float_const<From>::ZERO ? static_cast<To>(val) - static_cast<To>(1) : static_cast<To>(val);
284 }
